import time
import random
import matplotlib.pyplot as plt

#   
def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]

sizes = [5, 10, 50, 100, 500, 1000, 2000, 3000, 4000, 5000, 10000]
insertion_times = []
bubble_times = []

for size in sizes:
    test_array = [random.randint(0, 10000) for _ in range(size)]

    #   insertion sort
    array_copy = test_array[:]
    start_time = time.time()
    insertion_sort(array_copy)
    insertion_times.append(time.time() - start_time)

    #   bubble sort
    array_copy = test_array[:]
    start_time = time.time()
    bubble_sort(array_copy)
    bubble_times.append(time.time() - start_time)

plt.figure(figsize=(10, 6))
plt.plot(sizes, insertion_times, label=" ", marker='o')
plt.plot(sizes, bubble_times, label=" ", marker='o')
plt.title("   ")
plt.xlabel("  (n)")
plt.ylabel("  ()")
plt.legend()
plt.grid()
plt.show()
