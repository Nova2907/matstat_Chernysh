import cv2
import matplotlib.pyplot as plt
import os

image_paths = [
    r"C:\Amo pic\pic1.png",
    r"C:\Amo pic\pic2.png",
    r"C:\Amo pic\pic3.png",
    r"C:\Amo pic\pic4.png",
    r"C:\Amo pic\pic5.png"
]

for image_path in image_paths:
    img = cv2.imread(image_path)
    
    if img is not None:
        plt.imshow(cv2.cvtColor(img, cv2.COLOR_BGR2RGB))  # Перетворюємо з BGR на RGB
        plt.title(f"Displaying: {os.path.basename(image_path)}")
        plt.show()
    else:
        print(f"Failed to load image: {image_path}")
